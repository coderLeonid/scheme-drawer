//Определение максимального или минимального значения
//с использованием структур "выбор" и "обход".
//Вычислить x = min(min(a, b) - b, max(b, d) + c) 
#include <stdio.h>
#include<locale.h>
using namespace std;
int main()
{
	setlocale(LC_ALL, "Russian");
	int a, b, c, d,	//исходные данные
	max, min,		//промежуточные значения
	x;			//результат вычисления
	int i;		//вспомогательная переменная
	printf("Введите значения a, b, c, d: ");
	scanf_s("%i%i%i%i", &a, &b, &c, &d);
	//Решение задачи с использованием структуры "Выбор"
	if (a < b) min = a;	//определяем наименьшее значение между a и b
	else min = b;
	if (b < d) max = d;	//определяем наибольшее значение между b и d
	else max = b;
	if (min - b < max + c) x = min - b;//определяем наименьшее значение 	между min - b и max + c
	else x = max + c;
	//вывод результатов решения с использованием структуры "Выбор"
	printf("Использование структуры Выбор: x = %i\n", x);
	//Решение задачи с использованием структуры "Обход"
	min = a;	//определяем наименьшее значение между a и b
	if (min > b) min = b;
	max = b;	//определяем наибольшее значение между b и d
	if (max < d) max = d;
	x = min - b; //определяем наименьшее значение между max + c и min - b
	if (x > max + c) x = max + c;
	//вывод результатов решения с использованием структуры "Обход"
	printf("Использование структуры Обход: x = %i\n", x);
	printf("\n Повторить-1, Выход-2: ");
	scanf_s("%i", &i);
	if (i == 1) main();
	return 0;
}
